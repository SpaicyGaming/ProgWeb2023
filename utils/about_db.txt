------creazione tabelle:

CREATE TABLE USER_TYPES(USER_TYPE_ID INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1), USER_TYPE VARCHAR(128) NOT NULL, PRIMARY KEY(USER_TYPE_ID), UNIQUE(USER_TYPE));

CREATE TABLE USERS(USER_ID INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1), USERNAME VARCHAR(128) NOT NULL, MAIL VARCHAR(128) NOT NULL, NOME VARCHAR(128) NOT NULL, COGNOME VARCHAR(128) NOT NULL, NUMERO_DI_TELEFONO VARCHAR(128) NOT NULL, DATA_DI_NASCITA DATE NOT NULL, USER_TYPE INTEGER NOT NULL, UNIQUE(USERNAME), PRIMARY KEY(USER_ID), FOREIGN KEY(USER_TYPE) REFERENCES USER_TYPES(USER_TYPE_ID));

CREATE TABLE DONATIONS(DOANTION_ID INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1), DATA_DONAZIONE DATE NOT NULL, IMPORTO INTEGER NOT NULL);
